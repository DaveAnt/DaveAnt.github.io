UnityShader渐变纹理-卡通渲染

UnityShader渐变纹理-卡通渲染

April 28, 2020 字数 2846 极速模式

前言

渐变纹理通常是用来控制漫反射结果的,也可以得到插画、卡通风格的渲染效果,使用这种技术可以保证物体的的轮廓线相比于之前使用传统漫反射光照更加明显。

1.渐变纹理实战

最开始在渲染中使用纹理是为了定义物体的颜色,后来发现纹理其实可以用于存储任何表面属性,常见用法就是使用渐变纹理来控制漫反射光照的结果,在之前计算漫反射光照是,都是使用表面法线和光照方向的点积结果与材质的反射率相乘来得到表面漫反射光照。使用这种技术可以保证物体的轮廓相比于之前使用传统漫反射光照更加明显,而且能够提供多种色调变化。很多卡通、插画风格的渲染都使用这种技术。

Shader "Custom/Ramp Texture" {
	Properties {
		_Color ("Color Tint", Color) = (1, 1, 1, 1)
		_RampTex ("Ramp Tex", 2D) = "white" {}
		_Specular ("Specular", Color) = (1, 1, 1, 1)
		_Gloss ("Gloss", Range(8.0, 256)) = 20
	}
	SubShader {
		Pass { 
			Tags { "LightMode"="ForwardBase" }
		
			CGPROGRAM
			
			#pragma vertex vert
			#pragma fragment frag

			#include "Lighting.cginc"
			
			fixed4 _Color;
			sampler2D _RampTex;
			float4 _RampTex_ST;
			fixed4 _Specular;
			float _Gloss;
			
			struct a2v {
				float4 vertex : POSITION;
				float3 normal : NORMAL;
				float4 texcoord : TEXCOORD0;
			};
			
			struct v2f {
				float4 pos : SV_POSITION;
				float3 worldNormal : TEXCOORD0;
				float3 worldPos : TEXCOORD1;
				float2 uv : TEXCOORD2;
			};
			
			v2f vert(a2v v) {
				v2f o;
				o.pos = UnityObjectToClipPos(v.vertex);
				
				o.worldNormal = UnityObjectToWorldNormal(v.normal);
				
				o.worldPos = mul(unity_ObjectToWorld, v.vertex).xyz;
				
				o.uv = TRANSFORM_TEX(v.texcoord, _RampTex);
				
				return o;
			}
			
			fixed4 frag(v2f i) : SV_Target {
				fixed3 worldNormal = normalize(i.worldNormal);
				fixed3 worldLightDir = normalize(UnityWorldSpaceLightDir(i.worldPos));
				
				fixed3 ambient = UNITY_LIGHTMODEL_AMBIENT.xyz;
				
				// Use the texture to sample the diffuse color
				fixed halfLambert  = 0.5 * dot(worldNormal, worldLightDir) + 0.5;
				fixed3 diffuseColor = tex2D(_RampTex, fixed2(halfLambert, halfLambert)).rgb * _Color.rgb;
				
				fixed3 diffuse = _LightColor0.rgb * diffuseColor;
				
				fixed3 viewDir = normalize(UnityWorldSpaceViewDir(i.worldPos));
				fixed3 halfDir = normalize(worldLightDir + viewDir);
				fixed3 specular = _LightColor0.rgb * _Specular.rgb * pow(max(0, dot(worldNormal, halfDir)), _Gloss);
				
				return fixed4(ambient + diffuse + specular, 1.0);
			}
			
			ENDCG
		}
	} 
	FallBack "Specular"
}

首先使用内置TRANSFORM_TEX宏定义来计算经过平铺和偏移后的纹理坐标。接下来就是关键的片元着色器代码,很久之前提到过半兰伯特模型,通过这个理论把halfLambert的范围映射到[0,1]之间,之后使用它来构建纹理坐标,按老规矩渐变纹理_RampTex进行采样,然后把渐变纹理采样得到颜色和材质颜色_Color相乘,得到最终的漫反射颜色。这里使用黑-灰-白渐变纹理,通常用于卡通渲染的效果。

2.渐变纹理注意事项

将渐变纹理的Warp Mode设置为Clamp模式,以防止纹理进行采样时由于浮点数精度而造成的问题,下面给出Warp Mode分别为Repeat和Clamp模式的效果对比。 使用Repeat模式在高光区域会出现黑点,这是由于浮点精度造成的,使用fixed2(halfLambert,halfLambert)对渐变纹理进行采样时,虽然理论上halfLambert的值在[0,1]之间,但会有类似1.001这样值出现,如果使用Repeat模式,就会舍弃整数部分保留小数部分,导致最终的值接近0(即黑色),因此高光区会有黑点出现。

3.卡通渲染

单单使用渐变纹理控制面反射色调是完全不够,因为卡通风格的高光效果也和基本光照不同,在卡通风格中,模型的高光往往是一块分界明显的纯色区域。除了光照模型不同外,卡通风格通常还需要在物体边缘部分绘制轮廓(添加描边)。描边类型可以分成以下五种方式。

  • 基于观察角度和表面法线的轮廓。这种方式使用视角方向和表面法线的点乘结果来得到轮廓线的信息,这种方式简单快速在一个Pass中得到渲染结果,但局限性很大,很多模型渲染出来的描边效果都不尽人意。
  • 过程式几何轮廓。这种方式的核心式使用两个Pass渲染,第一个Pass渲染背面的面片,并使用某些技术让其轮廓可见;第二个Pass再正常渲染正面的面片。这种方式的优点在于快速有效,并且适用绝大多数表面平滑的模型,但是它不适合类似于立方体的平面模型。
  • 基于图像处理的轮廓线渲染,这种方式优点在于适用于任何类型的模型,但它也有自身的局限,一些升读和法线变化很小的的轮廓无法被检测出来,例如桌子上的纸张。
  • 基于轮廓边检测的轮廓线渲染。以上方式都无法实现轮廓线的独特风格渲染,如果精确检测出轮廓边就可渲染出独特的风格,检测公式很简单,检查和这条边相邻两个三角面片是否满足以下条件：(n0.v>0)≠(n1.v>0)
    其中n0和n1分别表示两个相邻三角面片的法向,v是从视角到该边任意顶点的方向,公式本质在于检查两个相邻的三角面片是否朝向正面、背面。它的缺点是除了实现相对复杂外,还会导致动画连贯性问题。
  • 最后一个就是混合上述几种渲染方式,比如首先找到精确的轮廓边,把模型和轮廓边渲染到纹理中,再使用图像处理方法识别出轮廓线,并在图像空间下进行风格化渲染。

卡通风格的高光往往是模型上一块块分界明显的纯色区域,所以可以使用以下方式去实现卡通高光：
float spec = dot(worldNormal,worldHalfDir);
Spec = step(threshold,spec);
这种简单粗暴的方式很出现明显的锯齿,所以公式可以调整成以下：
float spec = dot(worldNormal,worldHalfDir);
Spec = lerp(0,1,smoothstep(-w,w,spec - threshold));
上面用了很多内置函数,接下来给出表格。

函数 描述
step 实现阀值比较,参数1是参考值,参数2是待比较数值。如果参数1>=参数2返回1,不然返回0。
smoothstep w是很小的值,当spec - threshold小于-w返回0;大于w返回1。
lerp 0到1之间进行插值,从而达到平滑实现抗锯齿效果。 
Shader "Custom/Cartoon Shader" {
	Properties {
		_Color ("Color Tint", Color) = (1, 1, 1, 1)
		_MainTex ("Main Tex", 2D) = "white" {}
		_Ramp ("Ramp Texture", 2D) = "white" {}
		_Outline ("Outline", Range(0, 10)) = 0.1
		_OutlineColor ("Outline Color", Color) = (0, 0, 0, 1)
		_Specular ("Specular", Color) = (1, 1, 1, 1)
		_SpecularScale ("Specular Scale", Range(0, 0.1)) = 0.01
	}
    SubShader {
		Tags { "RenderType"="Opaque" "Queue"="Geometry"}
		
		Pass {
			NAME "OUTLINE"
			
			Cull Front
			
			CGPROGRAM
			
			#pragma vertex vert
			#pragma fragment frag
			
			#include "UnityCG.cginc"
			
			float _Outline;
			fixed4 _OutlineColor;
			
			struct a2v {
				float4 vertex : POSITION;
				float3 normal : NORMAL;
			}; 
			
			struct v2f {
			    float4 pos : SV_POSITION;
			};
			
			v2f vert (a2v v) {
				v2f o;
				
				float4 pos = mul(UNITY_MATRIX_MV, v.vertex); 
				float3 normal = mul((float3x3)UNITY_MATRIX_IT_MV, v.normal);  
				normal.z = -0.5;
				pos = pos + float4(normalize(normal), 0) * _Outline;
				o.pos = mul(UNITY_MATRIX_P, pos);
				
				return o;
			}
			
			float4 frag(v2f i) : SV_Target { 
				return float4(_OutlineColor.rgb, 1);               
			}
			
			ENDCG
		}
		
		Pass {
			Tags { "LightMode"="ForwardBase" }
			
			Cull Back
		
			CGPROGRAM
		
			#pragma vertex vert
			#pragma fragment frag
			
			#pragma multi_compile_fwdbase
		
			#include "UnityCG.cginc"
			#include "Lighting.cginc"
			#include "AutoLight.cginc"
			#include "UnityShaderVariables.cginc"
			
			fixed4 _Color;
			sampler2D _MainTex;
			float4 _MainTex_ST;
			sampler2D _Ramp;
			fixed4 _Specular;
			fixed _SpecularScale;
		
			struct a2v {
				float4 vertex : POSITION;
				float3 normal : NORMAL;
				float4 texcoord : TEXCOORD0;
				float4 tangent : TANGENT;
			}; 
		
			struct v2f {
				float4 pos : POSITION;
				float2 uv : TEXCOORD0;
				float3 worldNormal : TEXCOORD1;
				float3 worldPos : TEXCOORD2;
				SHADOW_COORDS(3)
			};
			
			v2f vert (a2v v) {
				v2f o;
				
				o.pos = UnityObjectToClipPos( v.vertex);
				o.uv = TRANSFORM_TEX (v.texcoord, _MainTex);
				o.worldNormal  = UnityObjectToWorldNormal(v.normal);
				o.worldPos = mul(unity_ObjectToWorld, v.vertex).xyz;
				
				TRANSFER_SHADOW(o);
				
				return o;
			}
			
			float4 frag(v2f i) : SV_Target { 
				fixed3 worldNormal = normalize(i.worldNormal);
				fixed3 worldLightDir = normalize(UnityWorldSpaceLightDir(i.worldPos));
				fixed3 worldViewDir = normalize(UnityWorldSpaceViewDir(i.worldPos));
				fixed3 worldHalfDir = normalize(worldLightDir + worldViewDir);
				
				fixed4 c = tex2D (_MainTex, i.uv);
				fixed3 albedo = c.rgb * _Color.rgb;
				
				fixed3 ambient = UNITY_LIGHTMODEL_AMBIENT.xyz * albedo;
				
				UNITY_LIGHT_ATTENUATION(atten, i, i.worldPos);
				
				fixed diff =  dot(worldNormal, worldLightDir);
				diff = (diff * 0.5 + 0.5) * atten;
				
				fixed3 diffuse = _LightColor0.rgb * albedo * tex2D(_Ramp, float2(diff, diff)).rgb;
				
				fixed spec = dot(worldNormal, worldHalfDir);
				fixed w = fwidth(spec) * 2.0;
				fixed3 specular = _Specular.rgb * lerp(0, 1, smoothstep(-w, w, spec + _SpecularScale - 1)) * step(0.0001, _SpecularScale);
				
				return fixed4(ambient + diffuse + specular, 1.0);
			}
		
			ENDCG
		}
	}
	FallBack "Diffuse"
}

其中Properties标签中,_Ramp用于控制漫反射的渐变纹理,_Outline用于控制轮廓线宽度,_OutlineColor对应轮廓线颜色,_Specular是高光反射颜色,_SpecularScale用于控制计算高光反射时的阀值。
在第一个Pass中使用了Cull指令把正面三角面片剔除,表面只渲染背面,为它定义了名字是描边是常见的效果,为了以后不重复编写此Pass,直接调用它的名字即可。第一个Pass对法线进行处理是为了尽可能避免背面扩张后顶点挡住正面的面片。 第二Pass剔除背面渲染,在顶点着色器中使用Unity内置的SHADOW_COORDS和TRSNDFER_DHSDOW来计算阴影所需要的各个变量。最后片元着色器中把计算光照需要的各个方向矢量进行归一化处理,然后计算材质的反射率albedo和环境光照ambient,接着使用内置的UNITY_LIGHT_ATTENUATION宏计算当前世界坐标下的阴影值,随后计算半兰伯特漫反射系数和阴影值相乘得到最终的漫反射系数,使用漫反射系数对渐变纹理_Ramp进行采样,将结果和材质反射率、光照颜色相乘作为最终的漫反射光照。 高光反射使用了fwidth对高光边界进行抗锯齿处理,并将计算而得到高光反射系数和高光反射颜色相乘,得到高光反射的光照部分。为了完全消除高光反射光照,使用了step(0.0001,_SpecularScale),将_SpecularScale设置为0即可。
