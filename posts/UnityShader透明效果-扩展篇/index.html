前言

在透明效果篇简单说明了透明测试和透明度混合简单实现,但有时会得到错误的效果,在最后扩展小节三里说明原因了,所以这里就尝试几种解决方案。

1.深度写入的半透明

说过解决的方案就是分割Mesh,理论上Mesh分割的越细就可以解决这种情况,但有时是不切实际的,这时就需要重新打开深度写入,让模型可以像半透明物体一样进行淡入淡出。
解决方案：使用两个Pass来渲染模型,第一个Pass开启深度写,但不输出颜色,目的仅仅是把该模型的深度值写入深度缓冲中;第二个Pass开启深度写入,由于上个Pass得到了逐像素的正确深度信息,按照这个像素深度进行透明渲染。

Shader "Custom/Alpha Blending With ZWrite" {
	Properties {
		_Color ("Color Tint", Color) = (1, 1, 1, 1)
		_MainTex ("Main Tex", 2D) = "white" {}
		_AlphaScale ("Alpha Scale", Range(0, 1)) = 1
	}
	SubShader {
		Tags {"Queue"="Transparent" "IgnoreProjector"="True" "RenderType"="Transparent"}
		
		// Extra pass that renders to depth buffer only
		Pass {
			ZWrite On
			ColorMask 0
		}
		
		Pass {
			Tags { "LightMode"="ForwardBase" }
			
			ZWrite Off
			Blend SrcAlpha OneMinusSrcAlpha
			
			CGPROGRAM
			
			#pragma vertex vert
			#pragma fragment frag
			
			#include "Lighting.cginc"
			
			fixed4 _Color;
			sampler2D _MainTex;
			float4 _MainTex_ST;
			fixed _AlphaScale;
			
			struct a2v {
				float4 vertex : POSITION;
				float3 normal : NORMAL;
				float4 texcoord : TEXCOORD0;
			};
			
			struct v2f {
				float4 pos : SV_POSITION;
				float3 worldNormal : TEXCOORD0;
				float3 worldPos : TEXCOORD1;
				float2 uv : TEXCOORD2;
			};
			
			v2f vert(a2v v) {
				v2f o;
				o.pos = UnityObjectToClipPos(v.vertex);
				
				o.worldNormal = UnityObjectToWorldNormal(v.normal);
				
				o.worldPos = mul(unity_ObjectToWorld, v.vertex).xyz;
				
				o.uv = TRANSFORM_TEX(v.texcoord, _MainTex);
				
				return o;
			}
			
			fixed4 frag(v2f i) : SV_Target {
				fixed3 worldNormal = normalize(i.worldNormal);
				fixed3 worldLightDir = normalize(UnityWorldSpaceLightDir(i.worldPos));
				
				fixed4 texColor = tex2D(_MainTex, i.uv);
				
				fixed3 albedo = texColor.rgb * _Color.rgb;
				
				fixed3 ambient = UNITY_LIGHTMODEL_AMBIENT.xyz * albedo;
				
				fixed3 diffuse = _LightColor0.rgb * albedo * max(0, dot(worldNormal, worldLightDir));
				
				return fixed4(ambient + diffuse, texColor.a * _AlphaScale);
			}
			
			ENDCG
		}
	} 
	FallBack "Transparent/VertexLit"
}

得到的效果图如下所示： 使用了新的渲染命令(ColorMask),在ShaderLab中ColorMask用于设置颜色通道的写入掩饰码,语义如下：

ColorMask RGB | A | 0 |其他任何R、G、B、A的组合

当ColorMask设为0时,意味着该Pass不写进入任何颜色通道,即不会输出任何颜色,下面给出ShaderLab支持的混合因子表格。

参数 描述
One 因子为1。
Zero 因子为0。
SrcColor 因子为源颜色值,当用于混合RGB等式时,使用SrcColor的RGB分量作为混合因子;当用于混合A的混合等式,使用SrcColor的A分量作为混合因子。
SrcAlpha 因子为源颜色的透明度值(A通道)。
DstColor 因子为源颜色值。当用于混合RGB通道的等式时,使用DstColor的RGB分量作为混合因子;当用于混合A的混合等式时,使用结果A分量作为混合因子。
DstAlpha 因子为目标颜色的透明度值(A通道)。
OneMinusSrcColor 因子为(1-源颜色),当用于混合RGB等式时,使用DstColor的RGB分量作为混合因子;当用于混合A通道的混合等式时,使用DstColor的A分量作为混合因子。
OneMinuseSrcAlpha 因子为(1-源颜色的透明度值)。
OneMinuseDstColor 因子为(1-目标颜色)。当用于混合RGB等式,使用结果的RGB分量作为混合因子,当用于混合A的混合等式时,使用结果A分量作为混合因子。
OneMinuseDstAlpha 因子(1-目标颜色的透明度值)。

如果想要在混合后,输出颜色的透明度值就是源颜色透明度,可以尝试以下命令：

Blend SrcAlpha OneMinusSrcAlpha, One Zero

源颜色和目标颜色与对应混合因子相乘后,都是把它们结果加起来输出颜色的,其实可以使用ShaderLab的BlendOp BlendOperation其他命令。

操作 描述
Add 将混合后的源颜色和目的颜色相加。默认的混合操作。使用的混合等式是：
Orgb=SrcFactor*Srbg+DstFactor*Drgb
Oa=SrcFactorA*Sa+DstFactorA*Da
Sub 用混合后的源颜色减去混合后的目标颜色。使用混合等式：
Orgb=SrcFactor*Srbg-DstFactor*Drgb
Oa=SrcFactorA*Sa-DstFactorA*Da
RevSub 用混合后的源颜色减去混合后的目标颜色。使用混合等式：
Orgb=DstFactor*Drgb-SrcFactor*Srbg
Oa=DstFactorA*Da-SrcFactorA*Sa
Min 使用源颜色和目标颜色较小值,是逐分量比较的。使用的混合等式：
Orgb=(min(Sr,Dr),min(Sb,Db),min(Sg,Dg),min(Sa,Da))
Max 使用源颜色和目标颜色较大值,是逐分量比较的。使用的混合等式：
Orgb=(max(Sr,Dr),max(Sb,Db),max(Sg,Dg),max(Sa,Da))

混合操作命令通常是与混合因子命令一起工作的,默认的混合操作种类是Add。需要注意：使用Max或Min混合操作时,混合因子实际上时不起任何作用的,它们仅会判断原始的源颜色和目的颜色之间比较结果。接下来就是常见的混合操作种类表格,Photoshop里有类似混合模式的效果。

混合操作种类 描述
Blend SrcAlpha OneMinusSrcAlpha 正常(normal),即透明度混合
Blend DstColor Zero 正片叠底(multiply),即相乘
Blend DstColor SrcColor 两倍相乘(2* multiply)
BlendOp Min 变暗(darken)
BlendOp Max 变亮(lightten)
BlendOne One OneMinusDstColor
Blend OneMinusDstColor One		 滤色
Blend One One 线性减淡

2.双面渲染的透明效果

实际上更加逼真的透明效果是可以看到模型内部整体和背面的,但上面实现的透明度效果无法出现这种效果,原因是默认情况下渲染引擎会剔除物体背面的渲染图元,而仅仅渲染了模型的正面,要得到很逼真的透明效果,必须执行CUll指令防止剔除,记住这样会导致图元成倍增加,因此有必须的需求时可以尝试双面渲染,不然从性能方面考虑建议不要这样做。

Cull Back|Front|Off
  • 透明度测试的双面渲染
Shader "Custom/Alpha Test With Both Side" {
	Properties {
		_Color ("Color Tint", Color) = (1, 1, 1, 1)
		_MainTex ("Main Tex", 2D) = "white" {}
		_Cutoff ("Alpha Cutoff", Range(0, 1)) = 0.5
	}
	SubShader {
		Tags {"Queue"="AlphaTest" "IgnoreProjector"="True" "RenderType"="TransparentCutout"}
		
		Pass {
			Tags { "LightMode"="ForwardBase" }
			Cull Off	
			CGPROGRAM
			
			#pragma vertex vert
			#pragma fragment frag
			
			#include "Lighting.cginc"
			
			fixed4 _Color;
			sampler2D _MainTex;
			float4 _MainTex_ST;
			fixed _Cutoff;
			
			struct a2v {
				float4 vertex : POSITION;
				float3 normal : NORMAL;
				float4 texcoord : TEXCOORD0;
			};
			
			struct v2f {
				float4 pos : SV_POSITION;
				float3 worldNormal : TEXCOORD0;
				float3 worldPos : TEXCOORD1;
				float2 uv : TEXCOORD2;
			};
			
			v2f vert(a2v v) {
				v2f o;
				o.pos = UnityObjectToClipPos(v.vertex);
				
				o.worldNormal = UnityObjectToWorldNormal(v.normal);
				
				o.worldPos = mul(unity_ObjectToWorld, v.vertex).xyz;
				
				o.uv = TRANSFORM_TEX(v.texcoord, _MainTex);
				
				return o;
			}
			
			fixed4 frag(v2f i) : SV_Target {
				fixed3 worldNormal = normalize(i.worldNormal);
				fixed3 worldLightDir = normalize(UnityWorldSpaceLightDir(i.worldPos));
				
				fixed4 texColor = tex2D(_MainTex, i.uv);

				clip (texColor.a - _Cutoff);
				
				fixed3 albedo = texColor.rgb * _Color.rgb;
				
				fixed3 ambient = UNITY_LIGHTMODEL_AMBIENT.xyz * albedo;
				
				fixed3 diffuse = _LightColor0.rgb * albedo * max(0, dot(worldNormal, worldLightDir));
				
				return fixed4(ambient + diffuse, 1.0);
			}
			
			ENDCG
		}
	} 
	FallBack "Transparent/Cutout/VertexLit"
}

  • 透明度混合的双面渲染
    透明度混合的双面渲染会更加复杂,自从关闭深度写入之后,一切混乱的开端就是开始了,因为无法保证同个物体的正面和背面图元按照顺序渲染。 为此双面渲染的工作分成两个Pass,第一个Pass只渲染背面,第二个渲染正面,由于Unity会安装顺序去执行各个Pass,从而达到渲染顺序正确得到需求。
Shader "Custom/Alpha Blend With Both Side" {
	Properties {
		_Color ("Color Tint", Color) = (1, 1, 1, 1)
		_MainTex ("Main Tex", 2D) = "white" {}
		_AlphaScale ("Alpha Scale", Range(0, 1)) = 1
	}
	SubShader {
		Tags {"Queue"="Transparent" "IgnoreProjector"="True" "RenderType"="Transparent"}
		
		Pass {
			Tags { "LightMode"="ForwardBase" }
			
			// First pass renders only back faces 
			Cull Front
			
			ZWrite Off
			Blend SrcAlpha OneMinusSrcAlpha
			
			CGPROGRAM
			
			#pragma vertex vert
			#pragma fragment frag
			
			#include "Lighting.cginc"
			
			fixed4 _Color;
			sampler2D _MainTex;
			float4 _MainTex_ST;
			fixed _AlphaScale;
			
			struct a2v {
				float4 vertex : POSITION;
				float3 normal : NORMAL;
				float4 texcoord : TEXCOORD0;
			};
			
			struct v2f {
				float4 pos : SV_POSITION;
				float3 worldNormal : TEXCOORD0;
				float3 worldPos : TEXCOORD1;
				float2 uv : TEXCOORD2;
			};
			
			v2f vert(a2v v) {
				v2f o;
				o.pos = UnityObjectToClipPos(v.vertex);
				
				o.worldNormal = UnityObjectToWorldNormal(v.normal);
				
				o.worldPos = mul(unity_ObjectToWorld, v.vertex).xyz;
				
				o.uv = TRANSFORM_TEX(v.texcoord, _MainTex);
				
				return o;
			}
			
			fixed4 frag(v2f i) : SV_Target {
				fixed3 worldNormal = normalize(i.worldNormal);
				fixed3 worldLightDir = normalize(UnityWorldSpaceLightDir(i.worldPos));
				
				fixed4 texColor = tex2D(_MainTex, i.uv);
				
				fixed3 albedo = texColor.rgb * _Color.rgb;
				
				fixed3 ambient = UNITY_LIGHTMODEL_AMBIENT.xyz * albedo;
				
				fixed3 diffuse = _LightColor0.rgb * albedo * max(0, dot(worldNormal, worldLightDir));
				
				return fixed4(ambient + diffuse, texColor.a * _AlphaScale);
			}
			
			ENDCG
		}
		
		Pass {
			Tags { "LightMode"="ForwardBase" }
			
			// Second pass renders only front faces 
			Cull Back
			
			ZWrite Off
			Blend SrcAlpha OneMinusSrcAlpha
			
			CGPROGRAM
			
			#pragma vertex vert
			#pragma fragment frag
			
			#include "Lighting.cginc"
			
			fixed4 _Color;
			sampler2D _MainTex;
			float4 _MainTex_ST;
			fixed _AlphaScale;
			
			struct a2v {
				float4 vertex : POSITION;
				float3 normal : NORMAL;
				float4 texcoord : TEXCOORD0;
			};
			
			struct v2f {
				float4 pos : SV_POSITION;
				float3 worldNormal : TEXCOORD0;
				float3 worldPos : TEXCOORD1;
				float2 uv : TEXCOORD2;
			};
			
			v2f vert(a2v v) {
				v2f o;
				o.pos = UnityObjectToClipPos(v.vertex);
				
				o.worldNormal = UnityObjectToWorldNormal(v.normal);
				
				o.worldPos = mul(unity_ObjectToWorld, v.vertex).xyz;
				
				o.uv = TRANSFORM_TEX(v.texcoord, _MainTex);
				
				return o;
			}
			
			fixed4 frag(v2f i) : SV_Target {
				fixed3 worldNormal = normalize(i.worldNormal);
				fixed3 worldLightDir = normalize(UnityWorldSpaceLightDir(i.worldPos));
				
				fixed4 texColor = tex2D(_MainTex, i.uv);
				
				fixed3 albedo = texColor.rgb * _Color.rgb;
				
				fixed3 ambient = UNITY_LIGHTMODEL_AMBIENT.xyz * albedo;
				
				fixed3 diffuse = _LightColor0.rgb * albedo * max(0, dot(worldNormal, worldLightDir));
				
				return fixed4(ambient + diffuse, texColor.a * _AlphaScale);
			}
			
			ENDCG
		}
	} 
	FallBack "Transparent/VertexLit"
}
