前言

使用Socket(Tcp)编写多人聊天室的服务器端,使用Unity进行简单聊天界面的客户端搭建。对Socket编程有恐惧的新手按照文章步骤来可以深入理解Socket和多线程相关知识点。

1.客户端开发

首先把客户端聊天界面搭建起来,因为聊天消息可能长度不一,需要不规则长度的滑动方式。可尝试使用博主近期开发的ScrollCircleMaker快速搭建聊天场景,然后加个输入框和发送按钮即可,具体效果图如下所示： 接下来尝试编写客户端通信代码,开启子线程进行网络消息接受、发送,具体代码如下:

using UnityEngine;
using UnityEngine.UI;
using System.Net;
using System.Net.Sockets;
using System.Text;
using System.Threading;
using UIPlugs.ScrollCircleMaker;

public class ChatNetwork : MonoBehaviour
{
    public const string m_ipAddress = "127.0.0.1";
    public const int m_port = 2345;

    private Socket m_clientSocket;
    private Thread m_reCeiveThread;
    private byte[] m_msgData = new byte[1024];//消息数据容器
    private string m_message;

    public InputField  m_input;
    public GameObject m_content;
    GroupChatMaker chatMaker;
    SynchronizationContext mainThreadSynContext;

    void Start()
    {
        m_input.text = "";
        chatMaker = (GroupChatMaker)m_content.GetComponent<ScrollCircleComponent>().baseMaker;
        mainThreadSynContext = SynchronizationContext.Current;
        ConnentToServer();

        //开启一个线程专门用于接收消息
        m_reCeiveThread = new Thread(ReceiveMessage);
        m_reCeiveThread.Start();
    }

    void Update()
    {
        if (!string.IsNullOrEmpty(m_message))
        {
            m_input.text += "\n" + m_message;
            Debug.Log("m_message:" + m_message);
            m_message = "";
        }
    }

    public void ConnentToServer()
    {
        m_clientSocket = new Socket(AddressFamily.InterNetwork, SocketType.Stream, ProtocolType.Tcp);
        Debug.Log("开始连接服务器");
        m_clientSocket.Connect(new IPEndPoint(IPAddress.Parse(m_ipAddress), m_port));
        Debug.Log("连接服务器执行完毕");
    }

    void SendMessageToServer(string message)
    {
        byte[] data = Encoding.UTF8.GetBytes(message);
        m_clientSocket.Send(data);
    }

    public void OnSendMessageClick()
    {
        string message = m_input.text;
        if (string.IsNullOrEmpty(message))
        {
            Debug.LogWarning("发送消息不能为空");
            return;
        }
        SendMessageToServer(message);
        m_input.text = "";
    }

    void ReceiveMessage()
    {
        while (true)
        {
            //在接受消息之前判断Socket是否连接
            if (m_clientSocket.Connected == false)
                break;
            int length = m_clientSocket.Receive(m_msgData);//这里会等待接收消息，程序暂停，只有接收到消息后才会继续执行
            m_message = Encoding.UTF8.GetString(m_msgData, 0, length);
            string chat_message = m_message;
            mainThreadSynContext.Send(new SendOrPostCallback((ss) =>
            {
                chatMaker.Helper.AddItem(chat_message);
            }), null);//通知主线程
        }
    }

    void OnDestroy()
    {
        m_clientSocket.Shutdown(SocketShutdown.Both);
        m_clientSocket.Close();
    }
}

需要注意调用插件接口时,必须回到主线程调用,Unity引擎相关操作为了确保线程安全都是不支持在子线程中调用的,博主使用SynchronizationContext的Send方法通知主线程调用插件接口。到这里就成功完成客户端的所有问题了,将ChatNetwork脚本随便挂载到场景中,指定对应按钮(OnSendMessageClick是按钮注册事件)和输入框,至于插件如何快速创建不规则单行滑动可参考Demo-8。

2.服务器开发

接下来尝试开发多人聊天服务器,通过死循环将请求的Socket保存到列表中,只要有客户端发送消息,服务器就广播列表中所有Socket,客户端接受到消息后做出相应的显示效果。这里把保存列表里的Socket稍微封装了一下,封装成ClientSocket类,里面多了监听消息函数和Socket链接状态。

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Net.Sockets;
using System.Net;
using System.Threading;

namespace ChatRoomServer
{
    class Program
    {
        public const string m_ipAddress = "127.0.0.1";
        public const int m_port = 2345;
        static Socket m_tcpServerServer;
        static List<ClientSocket> m_clientList = new List<ClientSocket>();
        static void Main(string[] args)
        {
            m_tcpServerServer = new Socket(AddressFamily.InterNetwork, SocketType.Stream, ProtocolType.Tcp);
            m_tcpServerServer.Bind(new IPEndPoint(IPAddress.Parse(m_ipAddress), m_port));           
            m_tcpServerServer.Listen(1000);//2.设置监听数量上限
            Console.WriteLine("server running...");
            while (true)
            {
                Socket tcpClientSocket = m_tcpServerServer.Accept();
                Console.WriteLine("a client is connected!");
                ClientSocket client = new ClientSocket(tcpClientSocket);
                m_clientList.Add(client);
                Thread.Sleep(1);
            }
        }

        public static void BroadCastMessage(string message)
        {
            if (m_clientList == null || m_clientList.Count == 0)
                return;
            byte[] data = Encoding.UTF8.GetBytes(message);
            //用来存储已经断开连接的客户端
            List<ClientSocket> notConnected = new List<ClientSocket>();
            foreach (var client in m_clientList)
            {
                if( client.IsConnected)
                    client.SendMessageToClient(data);
                else
                    notConnected.Add(client);
            }
            foreach (var item in notConnected)
            {
                m_clientList.Remove(item);
            }
        }
    }

    class ClientSocket
    {
        private Socket m_clientSocket;
        public bool IsConnected
        {
            get { return m_clientSocket.Connected; }
        }

        private byte[] m_data = new byte[1024];
        public ClientSocket(Socket s)
        {
            m_clientSocket = s;
            ThreadPool.QueueUserWorkItem(new WaitCallback(ReceiveMessage));
        }

        void ReceiveMessage(object a)
        {
            while (true)
            {
                if (m_clientSocket.Poll(10, SelectMode.SelectRead))
                {
                    m_clientSocket.Shutdown(SocketShutdown.Both);
                    m_clientSocket.Close();
                    break;
                }
                int length = m_clientSocket.Receive(m_data);
                string message = Encoding.UTF8.GetString(m_data, 0, length);
                IPEndPoint tmpSocketPoint = m_clientSocket.RemoteEndPoint as IPEndPoint;
                Console.WriteLine(tmpSocketPoint.Address.ToString() + ":" + message);
                Program.BroadCastMessage(message);
            }
        }

        public void SendMessageToClient(byte[] messageData)
        {
            m_clientSocket.Send(messageData);
        }
    }
}

服务器通过m_clientSocket.Poll试图读取客户端数据,如果10毫秒内读取不到,那么判断已经断开连接,直接清理掉这个客户端的Socket。
通过以上代码得知发送消息也是字节流,这样为什么不把接收到的字节流直接广播出去呢？而是要转换成字符串再转换回字节流？主要目的是剪裁字节流大小,原因有以下三点:

  • 1.每次都没有这么大数据传输(客户端会做限制,就像微信消息总是有发送上限的),直接发送1024的字节流增加服务器带宽压力。
  • 2.数组由于没有清理过,在接收消息后,数组后面元素可能存储上次数据,导致客户端接收消息混乱(比如说第一条消息比第二条消息长的时候)。
  • 3.客户端接受了无用的数据,实在没有必要。

3.扩展思考

在局域网中想尝试在不同机器上进行通信以上代码是不行的,因为代码客户端的请求直接写成本地IP(127.0.0.1),把相同程序放到不同机器上会导致链接服务器失败,可以尝试打开命令行输入以下指令查看本机所对应的局域网IP,然后将127.0.0.1全部改成所示的IP： 如何搭建出可以商用的多人聊天服务器框架？以上代码Socket处理性能很差,最大连接数估计不能超过100个,不然服务器压力会很大(虽然消息监听已经使用多线程),这里提供三个解决方案来提高链接数量。

  • 1.通过IOCP模型重写多人聊天服务器。
  • 2.采用分布式架构降低服务器压力。
  • 3.自行百度(怎么设计分布式高并发聊天室的服务器架构?),哈哈哈哈,是不是太臭不要脸了。
